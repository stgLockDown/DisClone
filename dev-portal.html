Create Your First Plugin in 30 Seconds

The Nexus SDK is loaded automatically. Just call NexusSDK.createPlugin() and you're live.

  id: 'my-awesome-plugin',
  name: 'My Awesome Plugin',
  version: '1.0.0',
  author: 'YourName',
  description: 'Adds cool features to the server'
}, (ctx) => {

  // Register a slash command
  ctx.registerCommand({
    name: 'hello',
    description: 'Say hello to the channel',
    execute: (context) => {
      return 'Hello from my plugin! ğŸ‘‹';

JavaScript â€” streaming-api.js 
// Start a screen share stream
const stream = await NexusSDK.streaming.startStream({
  channelId: 'vc-general',
  type: 'screen',        // 'screen' | 'window' | 'camera'
  quality: '1080p',       // 'auto' | '720p' | '1080p' | 'source'
  frameRate: 60,          // 15, 30, or 60 fps
  audio: true             // Include system audio
});

// Listen for viewers joining
NexusSDK.events.on('stream:viewer-joined', ({ streamId, userId }) => {
  console.log(`New viewer: ${userId}`);
});

// Change quality mid-stream
NexusSDK.streaming.setQuality(stream.id, '720p');

// Stop the stream
NexusSDK.streaming.stopStream(stream.id);
